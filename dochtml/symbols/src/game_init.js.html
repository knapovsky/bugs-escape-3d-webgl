 43 /**
 44 * Initialize HTML5 canvas element
 45 */
 46 function initCanvas(){
 47     
 48     // Get dom reference
 49     var canvas = document.getElementById("webgl-canvas");
 50 
 51     // Set width and heigth
 52     canvasWidth = canvas.width;
 53     canvasHeight = canvas.height;
 54 
 55     // Initialize
 56     initGL(canvas);
 57 }
 58 
 59 /**
 60 * This function initializes WebGL and sets
 61 * handlers to keyboard and mouse events.
 62 */
 63 function webGLStart() {
 64     
 65     var canvas = document.getElementById("webgl-canvas");
 66 
 67     canvasWidth = canvas.width;
 68     canvasHeight = canvas.height;
 69 
 70     // Initialize WebGL
 71     initGL(canvas);
 72     
 73     // Init Shaders
 74     if(glInitialized){
 75     	initShaders();
 76     }
 77 
 78     if(runWhenLoaded){
 79         loadLevel();
 80         logicBackup = clone(logic);
 81     }
 82 
 83     // Set handlers to keyboard
 84     document.onkeydown = handleKeyDown;
 85     document.onkeyup = handleKeyUp;
 86 
 87     // Set handlers to mouse
 88     canvas.onmousedown = handleMouseDown;
 89     document.onmouseup = handleMouseUp;
 90     
 91     // Set handlers to mouse move events
 92     $("#webgl-canvas").mousemove(function(event) {
 93         
 94         position = getPosition(event);
 95 		
 96 		currentPointerX = position.x;
 97 		currentPointerY = position.y;
 98 
 99     });
100     
101     // Mouse inside canvas - adding over class to canvas element
102     $("#webgl-canvas").mouseover(function(e)
103     {
104 		$(this).addClass("over");
105     });
106     // removing over class from canvas element
107 	$("#webgl-canvas").mouseout(function(e)
108     {
109 		$(this).removeClass("over");
110     });
111 
112     // Bind menu mouseClick, mouseOver, mouseRelease animations
113     // and actions
114     bindMenuToMouse();
115 
116     // Bind chooser button mouseClick, mouseOver, mouseRelease animations
117     // and actions
118     bindChooserButton();
119 
120     // Bind chooser sliders mouseClick, mouseOver, mouseRelease animations
121     // and actions
122     bindChooserSliders();
123 
124     // Bind levels mouseClick, mouseOver, mouseRelease animations
125     // and actions
126     if(glInitialized){
127     	bindLevels();
128     }
129 
130     // Bind audio player mouseClick, mouseOver, mouseRelease animations
131     // and actions
132     audioBindPlayerUI();
133 
134     // Bind icons mouseClick, mouseOver, mouseRelease animations
135     // and actions
136     bindIcons();
137 
138     // Bind mouse wheel in Firefox
139     if (window.addEventListener){
140 
141         // DOMMouseScroll is for mozilla.
142         canvas.addEventListener('DOMMouseScroll', handleWheel, false);
143     }
144 
145     // Set gl
146     if(glInitialized){
147     	gl.clearColor(0.0, 0.0, 0.0, 1.0);
148     	gl.enable(gl.DEPTH_TEST);
149     }
150 }
