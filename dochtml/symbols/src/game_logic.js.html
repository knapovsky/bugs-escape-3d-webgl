 44 * Used for representation of item in game field

 60 function LogicItem(structure) {

 62     /**
 63     * Item name
 64     */
 65     this.name = structure["name"];

 67     * Item GUID for various tests
 68     */
 69     this.guid = parseInt(structure["guid"]);

 71     * Item Class
 72     * Defines type of logic object
 73     *
 74     * Classes and several subclassed:
 75     * -------------------------------
 76     * 0. Nothing
 77     * 1. Bug
 78     * 2. Wall
 79     * 3. Nothing
 80     * 4. Exit
 81     * 5. Box
 82     * 6. Explosive
 83     * 7. Stone
 84     * 8. Doors
 85     * 9. Teleport
 86     * 10. Button
 87     * 11. Elevator
 88     * 12. Water
 89     * 13. Item
 90     *   0 Snorkel
 91     *   1 Backpack
 92     *   2 Nothing
 93     *   3 Food
 94     *   4 Dynamite
 95     *   5 Vitamine
 96     *   6 Light
 97     *   7 Weight
 98     *   8 Pickaxe
 99     * 14. Nothing
100     * 15. Sinking floor
101     * 16. Nothing
102     * 17. Nothing
103     * 18. Nothing
104     * 19. Inclined floor
105     * 20. Nothing
106     * 21. Invisible wall
107     */
108     this.itemClass = parseInt(structure["class"]);
109     /**
110     * Item subclass
111     * Detail of item class
112     * For example - Box
113     *   Light box - subclass 1
114     *   Heavy box - subclass 0
115     */
116     this.itemSubclass = parseInt(structure["subclass"]);
117     /**
118     * Position of item in game field
119     * [x, y, z]
120     */
121     this.position = structure["position"];
122     /**
123     * Rotation of item
124     * Used for inclined floor
125     */
126     this.rotation = parseInt(structure["rotation"]);
127     /**
128     * ID of item
129     * Used for connection between
130     * model and logic
131     */
132     this.id = parseInt(structure["container_id"]);
133     /**
134     * Not used yet
135     */
136     this.polyID = -1;
137     if(structure["poly_id"]){
138         this.polyID = parseInt(structure["poly_id"]);
139     }

141     /** ONLY FOR BUGS **/
142 
143     /**
144     * Direction of bug
145     */
146     this.direction = "left";
147     /**
148     * Indicates whether bug has vitamin
149     * in inventory.
150     */
151     this.vitamin = false;
152     /**
153     * Indicates whether bug has additional
154     * weight in inventory
155     */
156     this.weight = false;
157     /**
158     * Indicates whether bug has snorkel
159     * in inventory.
160     */
161     this.snorchl = false;
162 }

791     /**
792     * Weight of item is specified in this function.
793     * Used extensively.
794     * @param position Position of item
795     * @return Weight of item
796     */
797     this.getWeightOfObject = function(position) {
798         
799         // Is item outside game field?
800         if(this.isPositionOutside(position)){
801 
802             // Ouside of gamefield acts like static object
803             return 1000;
804         }
805         
806         // Reference to item
807         var tempObject = this.getObjectOnPosition(position);
808 
809         // Item not found 
810         if(tempObject == 0) return 0;
811 
812         // Returns weight according to itemClass and itemSubclass        
813         switch(tempObject.itemClass){
814             
815             // Nothing
816             case 0:
817                 return 0;
818                 break;
819             
820             // Bug
821             case 1:
822 
823                 // Bug with additional weight in inventory
824                 if(tempObject.weight){
825                     return 2
826                 }
827                 // Normal bug
828                 else{
829                     return 1;   
830                 }
831                 break;
832             
833             // Wall - static
834             case 2:
835                 return 1000;
836                 break;
837             
838             // Nothing
839             case 3:
840                 return 0;
841                 break;
842             
843             // Exit
844             case 4:
845                 return 1000;
846                 break;
847             
848             // Box
849             case 5:
850 
851                 // Heavy box
852                 if(tempObject.itemSubclass == 0){
853                     return 2;
854                 }
855                 // Light box
856                 else{
857                     return 1;  
858                 }
859                 break;
860             
861             // Explosive
862             case 6:
863                 return 2;
864                 break;
865             
866             // Rock
867             case 7:
868                 return 1000;
869                 break;
870             
871
