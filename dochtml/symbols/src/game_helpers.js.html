 42 /**

 47 function showFPS(){
 48 
 49     if(log){
 50         console.log("FPS:"+fps+ " clippedObjects: " + clippedObjects + " static: " + staticCounter + " dynamic " + dynamicCounter);
 51     }
 52 
 53     // Reinitialization
 54     fps = 0;
 55     clippedObjects = 0;
 56     staticCounter = 0;
 57     dynamicCounter = 0;
 58 
 59     // New timeout
 60     if(showFPSinConsole){
 61         fpsTimeoutID = setTimeout(showFPS, 1000);
 62     }
 63 }

 98 /**
 99 * Resets camera rotation and elevation, so that it
100 * is behing the selected bug.
101 */
102 function resetCamera(){
103 
104     switch(logic.dynamicItems[logic.selectedBugID].direction){
105 
106         case "up":
107             currentRotation = 90;
108             break;
109         case "right":
110             currentRotation = 180;
111             break;
112         case "down":
113             currentRotation = 270;
114             break;
115         case "left":
116             currentRotation = 360;
117             break;
118     }
119 
120     currentRotation -= 90;
121 
122     currentElevation = 15;
123 }

134 /**
135 * Determines if GUID belongs to static or dynamic object
136 * @param guid GUID attribute of object
137 * @return If object is static then true, else false
138 */
139 function isObjectStatic( guid ){
140 	
141 	var itemClass = Math.floor(guid / 1000);
142 	
143 	return (itemClass == 2 || itemClass == 9 || itemClass == 17 || itemClass == 19 || itemClass == 20 || itemClass == 12 || itemClass == 4);
144 }
145 
146 /** 
147 * Determines if GUID belong to bug 
148 * @param guid GUID attribute of object
149 * @return true if guid belongs to bug, else false
150 */
151 function isObjectBug( guid ){
152 	
153 	var itemClass = Math.floor(guid / 1000);
154 	
155 	return (itemClass == 1);
156 }

203 /**
204 * Clones object 
205 * @param obj Object
206 * @return Copied object
207 */
208 function clone(obj){
209 
210     if(obj == null || typeof(obj) != 'object')
211         return obj;
212 
213     var temp = obj.constructor();
214 
215     // Copy it key by key
216     for(var key in obj)
217         temp[key] = clone(obj[key]);
218 
219     return temp;
220 }

253 function toArray(strg){
254     strg = strg.replace(/left|top/g,'0px');
255     strg = strg.replace(/right|bottom/g,'100%');
256     strg = strg.replace(/([0-9\.]+)(\s|\)|$)/g,"$1px$2");
257     var res = strg.match(/(-?[0-9\.]+)(px|\%|em|pt)\s(-?[0-9\.]+)(px|\%|em|pt)/);
258     return [parseFloat(res[1],10),res[2],parseFloat(res[3],10),res[4]];
259 }
