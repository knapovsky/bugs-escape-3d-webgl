 42 /**
 43 * This function is called in every frame.
 44 * Used only when browser window visible
 45 */
 46 function tick() {
 47 
 48     // Plan next tick
 49     requestAnimFrame(tick);
 50 
 51     // Handle mouse pointer
 52     handlePointer();
 53 
 54     // Draw next frame
 55     drawScene();
 56 
 57     // Do animation
 58     //animate();
 59 }
 60 
 61 /**
 62 * This function is used for rendering scene throught WebGL
 63 * 1. set mvMatrix
 64 * 2. set frustrum
 65 * 3. draw floor
 66 * 4. draw static objects
 67 * 5. draw dynamic objects
 68 * 6. draw blended objects
 69 */
 70 function drawScene() {
 71     
 72     if(!draw) return;
 73 
 74     // Array for blended objects
 75     var blendedObjects = [];
 76 
 77     // FPS counter
 78     fps++;
 79 
 80     // Set viewport    
 81     gl.viewport(0, 0, gl.viewportWidth, gl.viewportHeight);
 82 
 83     // Clear buffers
 84     gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
 85 
 86     // Set perspective
 87     mat4.perspective(projectionAngle, gl.viewportWidth / gl.viewportHeight, 0.1, 400.0, pMatrix);
 88     
 89     // Camera Matrix
 90     var cameraMatrix = mat4.create();
 91     mat4.identity(cameraMatrix);
 92 
 93     // Model-View Matrix
 94     var mvMatrixStatic = mat4.create();
 95     mat4.identity(mvMatrixStatic);
 96 
 97     //
 98     // Set camera matrix
 99     //
100 
101     // Center camera - rotation and elevation on
102     if(cameraMode == "centerCamera" || logic.gameOver){
103         
104         var Y = 0;
105         if(moveAlongYAxis){
106             Y = logic.centerY;
107         }
108 
109         // X, Z translation
110         mat4.translate(cameraMatrix, [logic.centerX, Y, logic.centerZ]);  
111 
112         // Rotation
113         mat4.rotate(cameraMatrix, degToRad(-currentRotation), [0, 1, 0]);
114 
115         // Elevation
116         if(useTopCamera){
117             mat4.rotate(cameraMatrix, degToRad(-90), [1, 0, 0]);
118         }
119         else{
120             mat4.rotate(cameraMatrix, degToRad(-currentElevation), [1, 0, 0]);  
121         }
122 
123         // Distance from scene
124         mat4.translate(cameraMatrix, [0, 0, distanceFromScene]);
125 
126     }
127 
128     // FPS Camera - rotation, elevation off
129     else if(cameraMode == "fpsCamera"){
130 
131         var bugPosition = logic.dynamicItems[logic.selectedBugID].position;
132 
133         // Translate camera to bugs position
134         mat4.translate(cameraMatrix, [ -logic.startX - (bugPosition[0] * logic.itemSize + 1),  
135                                         logic.startY + (bugPosition[2] + 0.7), 
136                                         logic.startZ + (bugPosition[1] * logic.itemSize + 1)]);
137         
138         // No distance
139         var distance = 0;
140         var rotation;
141 
142         // Camera rotation according to bugs rotation
143         switch(logic.dynamicItems[logic.selectedBugID].direction){
144 
145             case "up":
146                 rotation = 90;
147                 break;
148             case "right":
149                 rotation = 180;
150                 break;
151             case "down":
152                 rotation = 270;
153                 break;
154             case "left":
155                 rotation = 360;
156                 break;
157         }
158 
159         // Hack :)
160         rotation += 180;
161 
162         // Prepared for FPS camera rotation
163         /*if(useRotationInFPS){
164             mat4.rotate(cameraMatrix, degToRad(-currentRotation), [0, 1, 0]);
165             mat4.rotate(cameraMatrix, degToRad(-currentElevation), [1, 0, 0]);  
166             mat4.translate(cameraMatrix, [0, 0, distance]); 
167         }*/
168         //else{
169             mat4.rotate(cameraMatrix, degToRad(-rotation), [0, 1, 0]);
170             mat4.rotate(cameraMatrix, degToRad(-15), [1, 0, 0]);  
171             mat4.translate(cameraMatrix, [0, 0, distance]);
172         //}
173 
174     }
175 
176     // Follow camera - rotation, elevation on
177     else if(cameraMode == "followCamera"){
178 
179         var bugPosition = logic.dynamicItems[logic.selectedBugID].position;
180 
181         // Translate to bugs position
182         mat4.translate(cameraMatrix, [ -logic.startX - (bugPosition[0] * logic.itemSize) - 1,  
183                                         logic.startY, 
184                                         logic.startZ + (bugPosition[1] * logic.itemSize) + 1]);
185 
186         // Use current rotation
187         mat4.rotate(cameraMatrix, degToRad(-currentRotation), [0, 1, 0]);
188 
189         // Use current elevation
190         if(useTopCamera){
191             mat4.rotate(cameraMatrix, degToRad(-90), [1, 0, 0]);
192         }
193         else{
194             mat4.rotate(cameraMatrix, degToRad(-currentElevation), [1, 0, 0]);  
195         }
196 
197         // Move from scene
198         mat4.translate(cameraMatrix, [0, 0, distanceFromScene]);
199     }
200 
201     // Behind camera - rotation, elevation off
202     else if(cameraMode == "behindCamera"){
203 
204         var bugPosition = logic.dynamicItems[logic.selectedBugID].position;
205 
206         // Translate to bugs position
207         mat4.translate(cameraMatrix, [ -logic.startX - (bugPosition[0] * logic.itemSize + 1),  
208                                         logic.startY + (bugPosition[2]), 
209                                         logic.startZ + (bugPosition[1] * logic.itemSize + 1)]);
210         
211         var distance = 3;
212         var rotation;
213 
214         // Camera rotation according to bug current rotation
215         switch(logic.dynamicItems[logic.selectedBugID].direction){
216 
217             case "up":
218                 rotation = 90;
219                 break;
220             case "right":
221                 rotation = 180;
222                 break;
223             case "down":
224                 rotation = 270;
225                 break;
226             case "left":
227                 rotation = 360;
228                 break;
229         }
230 
231         // HACK :D
232         rotation += 180;
233 
234         // Make transformations
235         mat4.rotate(cameraMatrix, degToRad(-rotation), [0, 1, 0]);
236         mat4.rotate(cameraMatrix, degToRad(-20), [1, 0, 0]);  
237         mat4.translate(cameraMatrix, [0, 0, distance]); 
238     }
239     
240     // Set Model-View matrix
241     mat4.identity(mvMatrix);
242     mat4.inverse(cameraMatrix, cameraMatrix);
243     mat4.multiply(mvMatrix, cameraMatrix, mvMatrix);
244     mat4.multiply(mvMatrix, mvMatrixStatic, mvMatrixStatic);
245 
246     //
247     // Frustrum culling
248     // Not working yet
249     //
250     if(useClipping){
251 
252         var clip = [];
253 
254         /* Combine the two matrices (multiply projection by modelview) */
255         clip[ 0] = mvMatrix[ 0] * pMatrix[ 0] + mvMatrix[ 1] * pMatrix[ 4] + mvMatrix[ 2] * pMatrix[ 8] + mvMatrix[ 3] * pMatrix[12];
256         clip[ 1] = mvMatrix[ 0] * pMatrix[ 1] + mvMatrix[ 1] * pMatrix[ 5] + mvMatrix[ 2] * pMatrix[ 9] + mvMatrix[ 3] * pMatrix[13];
257         clip[ 2] = mvMatrix[ 0] * pMatrix[ 2] + mvMatrix[ 1] * pMatrix[ 6] + mvMatrix[ 2] * pMatrix[10] + mvMatrix[ 3] * pMatrix[14];
258         clip[ 3] = mvMatrix[ 0] * pMatrix[ 3] + mvMatrix[ 1] * pMatrix[ 7] + mvMatrix[ 2] * pMatrix[11] + mvMatrix[ 3] * pMatrix[15];
259 
260         clip[ 4] = mvMatrix[ 4] * pMatrix[ 0] + mvMatrix[ 5] * pMatrix[ 4] + mvMatrix[ 6] * pMatrix[ 8] + mvMatrix[ 7] * pMatrix[12];
261         clip[ 5] = mvMatrix[ 4] * pMatrix[ 1] + mvMatrix[ 5] * pMatrix[ 5] + mvMatrix[ 6] * pMatrix[ 9] + mvMatrix[ 7] * pMatrix[13];
262         clip[ 6] = mvMatrix[ 4] * pMatrix[ 2] + mvMatrix[ 5] * pMatrix[ 6] + mvMatrix[ 6] * pMatrix[10] + mvMatrix[ 7] * pMatrix[14];
263         clip[ 7] = mvMatrix[ 4] * pMatrix[ 3] + mvMatrix[ 5] * pMatrix[ 7] + mvMatrix[ 6] * pMatrix[11] + mvMatrix[ 7] * pMatrix[15];
264 
265         clip[ 8] = mvMatrix[ 8] * pMatrix[ 0] + mvMatrix[ 9] * pMatrix[ 4] + mvMatrix[10] * pMatrix[ 8] + mvMatrix[11] * pMatrix[12];
266         clip[ 9] = mvMatrix[ 8] * pMatrix[ 1] + mvMatrix[ 9] * pMatrix[ 5] + mvMatrix[10] * pMatrix[ 9] + mvMatrix[11] * pMatrix[13];
267         clip[10] = mvMatrix[ 8] * pMatrix[ 2] + mvMatrix[ 9] * pMatrix[ 6] + mvMatrix[10] * pMatrix[10] + mvMatrix[11] * pMatrix[14];
268         clip[11] = mvMatrix[ 8] * pMatrix[ 3] + mvMatrix[ 9] * pMatrix[ 7] + mvMatrix[10] * pMatrix[11] + mvMatrix[11] * pMatrix[15];
269 
270         clip[12] = mvMatrix[12] * pMatrix[ 0] + mvMatrix[13] * pMatrix[ 4] + mvMatrix[14] * pMatrix[ 8] + mvMatrix[15] * pMatrix[12];
271         clip[13] = mvMatrix[12] * pMatrix[ 1] + mvMatrix[13] * pMatrix[ 5] + mvMatrix[14] * pMatrix[ 9] + mvMatrix[15] * pMatrix[13];
272         clip[14] = mvMatrix[12] * pMatrix[ 2] + mvMatrix[13] * pMatrix[ 6] + mvMatrix[14] * pMatrix[10] + mvMatrix[15] * pMatrix[14];
273         clip[15] = mvMatrix[12] * pMatrix[ 3] + mvMatrix[13] * pMatrix[ 7] + mvMatrix[14] * pMatrix[11] + mvMatrix[15] * pMatrix[15];
274 
275         var frustum = [];
276         /* Extract the numbers for the RIGHT plane */
277         frustum[0] = [];
278         frustum[0][0] = clip[ 3] - clip[ 0];
279         frustum[0][1] = clip[ 7] - clip[ 4];
280         frustum[0][2] = clip[11] - clip[ 8];
281         frustum[0][3] = clip[15] - clip[12];
282 
283         if(normalizeFrustum){
284 
285             /* Normalize the result */
286             var t = Math.sqrt( frustum[0][0] * frustum[0][0] + frustum[0][1] * frustum[0][1] + frustum[0][2] * frustum[0][2] );
287             frustum[0][0] /= t;
288             frustum[0][1] /= t;
289             frustum[0][2] /= t;
290             frustum[0][3] /= t;
291         }
292 
293         /* Extract the numbers for the LEFT plane */
294         frustum[1] = [];
295         frustum[1][0] = clip[ 3] + clip[ 0];
296         frustum[1][1] = clip[ 7] + clip[ 4];
297         frustum[1][2] = clip[11] + clip[ 8];
298         frustum[1][3] = clip[15] + clip[12];
299 
300         if(normalizeFrustum){
301 
302             /* Normalize the result */
303             t = Math.sqrt( frustum[1][0] * frustum[1][0] + frustum[1][1] * frustum[1][1] + frustum[1][2] * frustum[1][2] );
304             frustum[1][0] /= t;
305             frustum[1][1] /= t;
306             frustum[1][2] /= t;
307             frustum[1][3] /= t;
308         }
309 
310         /* Extract the BOTTOM plane */
311         frustum[2] = [];
312         frustum[2][0] = clip[ 3] + clip[ 1];
313         frustum[2][1] = clip[ 7] + clip[ 5];
314         frustum[2][2] = clip[11] + clip[ 9];
315         frustum[2][3] = clip[15] + clip[13];
316 
317         if(normalizeFrustum){
318 
319             /* Normalize the result */
320             t = Math.sqrt( frustum[2][0] * frustum[2][0] + frustum[2][1] * frustum[2][1] + frustum[2][2] * frustum[2][2] );
321             frustum[2][0] /= t;
322             frustum[2][1] /= t;
323             frustum[2][2] /= t;
324             frustum[2][3] /= t;
325         }
326 
327         /* Extract the TOP plane */
328         frustum[3] = [];
329         frustum[3][0] = clip[ 3] - clip[ 1];
330         frustum[3][1] = clip[ 7] - clip[ 5];
331         frustum[3][2] = clip[11] - clip[ 9];
332         frustum[3][3] = clip[15] - clip[13];
333 
334         if(normalizeFrustum){
335 
336             /* Normalize the result */
337             t = Math.sqrt( frustum[3][0] * frustum[3][0] + frustum[3][1] * frustum[3][1] + frustum[3][2] * frustum[3][2] );
338             frustum[3][0] /= t;
339             frustum[3][1] /= t;
340             frustum[3][2] /= t;
341             frustum[3][3] /= t;
342         }
343 
344         /* Extract the FAR plane */
345         frustum[4] = [];
346         frustum[4][0] = clip[ 3] - clip[ 2];
347         frustum[4][1] = clip[ 7] - clip[ 6];
348         frustum[4][2] = clip[11] - clip[10];
349         frustum[4][3] = clip[15] - clip[14];
350 
351         if(normalizeFrustum){
352 
353             /* Normalize the result */
354             t = Math.sqrt( frustum[4][0] * frustum[4][0] + frustum[4][1] * frustum[4][1] + frustum[4][2] * frustum[4][2] );
355             frustum[4][0] /= t;
356             frustum[4][1] /= t;
357             frustum[4][2] /= t;
358             frustum[4][3] /= t;
359         }
360 
361         /* Extract the NEAR plane */
362         frustum[5] = [];
363         frustum[5][0] = clip[ 3] + clip[ 2];
364         frustum[5][1] = clip[ 7] + clip[ 6];
365         frustum[5][2] = clip[11] + clip[10];
366         frustum[5][3] = clip[15] + clip[14];
367 
368         if(normalizeFrustum){
369 
370             /* Normalize the result */
371             t = Math.sqrt( frustum[5][0] * frustum[5][0] + frustum[5][1] * frustum[5][1] + frustum[5][2] * frustum[5][2] );
372             frustum[5][0] /= t;
373             frustum[5][1] /= t;
374             frustum[5][2] /= t;
375             frustum[5][3] /= t;
376         }
377     }
378 
379     // Set specular
380     gl.uniform1i(shaderProgram.showSpecularHighlightsUniform, useSpecular);
381     gl.uniform1f(shaderProgram.materialShininessUniform, 30);
382 
383     // Set lightning
384     gl.uniform1i(shaderProgram.useLightingUniform, useLightning);
385 
386     //
387     // Draw floor
388     //
389     if(drawOnlyGameField && drawFloor && !useOpacity && !drawJustBug){
390 
391         mvPushMatrix();
392 
393         // Translate floor
394         mat4.translate(mvMatrix, [-(logic.startX + (logic.itemSize * logic.sizeX) - 1), 
395                                     logic.startY, 
396                                    (logic.startZ + logic.sizeZ * logic.itemSize - 1)]);
397 
398         // Load vertices
399         gl.bindBuffer(gl.ARRAY_BUFFER, logic.floorPlaneBuffer);
400         gl.vertexAttribPointer(shaderProgram.vertexPositionAttribute, 
401                                logic.floorPlaneBuffer.itemSize, gl.FLOAT, false, 0, 0);
402 
403         // Load texture coords
404         gl.bindBuffer(gl.ARRAY_BUFFER, logic.floorPlaneTextureCoordBuffer);
405         gl.vertexAttribPointer(shaderProgram.textureCoordAttribute, 
406                                logic.floorPlaneTextureCoordBuffer.itemSize, 
407                                gl.FLOAT, false, 0, 0);
408 
409         // Load texture
410         gl.uniform1i(shaderProgram.useTexturesUniform, true);
411         gl.activeTexture(gl.TEXTURE0);
412         gl.bindTexture(gl.TEXTURE_2D, materials["floor_texture"].textures[0]);
413         gl.uniform1i(shaderProgram.samplerUniform, 0);       
414         
415         // Set Matrices
416         setMatrixUniforms();
417         
418         // Draw floor
419         if(renderMode == "triangles"){
420             gl.drawArrays(gl.TRIANGLE_STRIP, 0, logic.floorPlaneBuffer.numItems);
421         }
422         else if(renderMode == "points"){
423             gl.drawArrays(gl.POINTS, 0, logic.floorPlaneBuffer.numItems);
424         }
425         else if(renderMode == "lineStrip"){
426             gl.drawArrays(gl.LINE_STRIP, 0, logic.floorPlaneBuffer.numItems);
427         }
428 
429         mvPopMatrix();
430     }
431     
432     //
433     // DRAW STATIC
434     //
435     for (staticObject in staticObjects) {
436         
437         if(drawJustBug){
438             continue;
439         }
440 
441         // FOR LIGHTMAPS
442         if(staticObjects[staticObject].isItPoly){
443             //gl.uniform1i(shaderProgram.useLightingUniform, true);
444         }
445         else{
446             //gl.uniform1i(shaderProgram.useLightingUniform, false);
447         }
448 
449         // Draw only game field
450         if(drawOnlyGameField && !staticObjects[staticObject].gameField){
451 
452             continue;
453         }
454 
455         // Drawing models from container
456         for (object in staticObjects[staticObject].objects) {
457 
458             // Transformed center of model
459             var newCenter = [0, 0, 0];
460 
461             // Transform center according to bounding box
462             mat4.multiplyVec3(mvMatrix, staticObjects[staticObject].objects[object].center, newCenter);
463 
464             // Frustrum culling
465             if(useClipping){
466 
467                 var clipped = false;
468                 
469                 var thisObject = staticObjects[staticObject].objects[object];
470                 var envelope = thisObject.envelopePoints;
471 
472                 // Is this model inside clipping frustrum?
473                 var outsidePoints = 0;
474                 var i;
475                 for( i = 0; i < 8; i++){
476 
477                     var p;
478                     for( p = 0; p < 6; p++ ){
479 
480                         var testedVector = [0, 0, 0];
481                         mat4.multiplyVec3(mvMatrix, envelope[i], testedVector);
482                         if( frustum[p][0] * testedVector[0] 
483                             + frustum[p][1] * testedVector[1] 
484                             + frustum[p][2] * testedVector[2] 
485                             + frustum[p][3] <= clippingThreshold ){
486 
487                             // Point outside
488                             outsidePoints++;
489                             
490                         }
491                     }   
492 
493                 }
494 
495                 // If all point of bounding box outside clipping frustrum,
496                 // there is no reason to draw this model
497                 if(outsidePoints >= envelope.length){
498 
499                     clippedObjects++;
500                     continue;
501                 }
502             }
503 
504             // Blending enabled
505             if(useOpacity && staticObjects[staticObject].gameField){
506                 
507                 // Add model into array and continue
508                 blendedObjects.push(new BlendedObject(staticObjects[staticObject].objects[object], 
509                                                       newCenter[2], 
510                                                       object, 
511                                                       true));
512                 continue;
513             }
514 
515             // Set pipeline
516             gl.disable(gl.BLEND);
517             gl.enable(gl.DEPTH_TEST);
518             gl.uniform1f(shaderProgram.alphaUniform, 1);
519 
520             // Load vertices
521             gl.bindBuffer(gl.ARRAY_BUFFER, staticObjects[staticObject].objects[object].vertexPositionBuffer);
522             gl.vertexAttribPointer(shaderProgram.vertexPositionAttribute, 
523                                    staticObjects[staticObject].objects[object].vertexPositionBuffer.itemSize, 
524                                    gl.FLOAT, false, 0, 0);
525 
526             // Load texture coords
527             gl.bindBuffer(gl.ARRAY_BUFFER, staticObjects[staticObject].objects[object].vertexTextureCoordBuffer);
528             gl.vertexAttribPointer(shaderProgram.textureCoordAttribute, 
529                                    staticObjects[staticObject].objects[object].vertexTextureCoordBuffer.itemSize, 
530                                    gl.FLOAT, false, 0, 0);
531 
532             // If textures enabled, load textures
533             if(useTextures){
534 
535                 gl.uniform1i(shaderProgram.useTexturesUniform, true);
536                 gl.activeTexture(gl.TEXTURE0);
537                 gl.bindTexture(gl.TEXTURE_2D, materials[staticObjects[staticObject].objects[object].material].textures[0]);
538                 gl.uniform1i(shaderProgram.samplerUniform, 0);
539             }
540             else{
541 
542                 gl.uniform1i(shaderProgram.useTexturesUniform, false);
543             }
544             
545             // If lightmaps enabled, load lightmaps
546             if(useLightmaps && staticObjects[staticObject].isItPoly){
547                 
548                 gl.uniform1i(shaderProgram.useLightmapUniform, 1);
549 
550                 gl.bindBuffer(gl.ARRAY_BUFFER, staticObjects[staticObject].objects[object].vertexLightmapCoordBuffer);
551                 gl.vertexAttribPointer(shaderProgram.lightmapCoordAttribute, 
552                                        staticObjects[staticObject].objects[object].vertexLightmapCoordBuffer.itemSize, 
553                                        gl.FLOAT, false, 0, 0);
554                 
555                 gl.activeTexture(gl.TEXTURE1);
556                 gl.bindTexture(gl.TEXTURE_2D, lightmaps[staticObjects[staticObject].polyID].lightmap );
557                 gl.uniform1i(shaderProgram.lightmapSamplerUniform, 1);
558             }
559 
560             // Load normals
561             gl.bindBuffer(gl.ARRAY_BUFFER, staticObjects[staticObject].objects[object].vertexNormalBuffer);
562             gl.vertexAttribPointer(shaderProgram.vertexNormalAttribute, staticObjects[staticObject].objects[object].vertexNormalBuffer.itemSize, gl.FLOAT, false, 0, 0);
563 
564             // Load indices
565             gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, staticObjects[staticObject].objects[object].vertexIndexBuffer);
566 
567             // Set matrices
568             setMatrixUniforms();
569 
570             // Draw model according to selected mode
571             if(renderMode == "triangles"){
572                 gl.drawElements(gl.TRIANGLES, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
573             }
574             else if(renderMode == "points"){
575                 gl.drawElements(gl.POINTS, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
576             }
577             else if(renderMode == "lineStrip"){
578                 gl.drawElements(gl.LINE_STRIP, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
579             }
580             else if(renderMode == "lineLoop"){
581                 gl.drawElements(gl.LINE_LOOP, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
582             }
583             else if(renderMode == "lines"){
584                 gl.drawElements(gl.LINES, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
585             }
586             else if(renderMode == "triangleStrip"){
587                 gl.drawElements(gl.TRIANGLE_STRIP, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
588             }
589             else if(renderMode == "triangleFan"){
590                 gl.drawElements(gl.TRIANGLE_FAN, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
591             }
592             else if(renderMode == "quadStrip"){
593                 gl.drawElements(gl.QUAD_STRIP, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
594             }
595             else if(renderMode == "quads"){
596                 gl.drawElements(gl.QUADS, staticObjects[staticObject].objects[object].vertexIndexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
597             }
