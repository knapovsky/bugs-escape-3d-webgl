 43 /**
 44 * Callback called when texture has been loaded
 45 * Updates loading progress bar
 46 */
 47 function textureLoaded(){
 48 
 49     // Next texture
 50     loadedTextureCount++;
 51 
 52     // New progressbar value
 53     var value = (loadedTextureCount / textureCount) * 100;
 54     $("#progressbar").progressbar({
 55 
 56         value: (loadedJSONValue + value)
 57     });
 58 
 59     // When fully loaded, hide progressbar
 60     if(loadedTextureCount == textureCount){
 61 
 62         $("#progressbar").stop().animate({'opacity':'0.0'}, 400);
 63         $("#notification").animate({'opacity':'0.0'}, 400);
 64 
 65         // Level loaded
 66         loaded = true;
 67 
 68         showTime = false;
 69     }
 70 }

 72 /** 
 73 * Handles loaded texture...as expected :D
 74 * Creates texture buffers, sets filtering, ...
 75 * @param texture Texture object including texture image
 76 */
 77 function handleLoadedTexture(texture) {
 78 
 79     /**
 80     * There is different coordination system in WebGL, so every
 81     * texture has to be fliiped aroud Y axis.
 82     * Point [0, 0] is top left corner of texture.
 83     */
 84     gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true);
 85 
 86     // You have to bind texture before working with it
 87     gl.bindTexture(gl.TEXTURE_2D, texture);
 88 
 89     // Loading texture into GPU
 90     gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, texture.image);
 91 
 92     // Magnify filtering - how texture acts when being upscaled
 93     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR); 
 94 
 95     // How texture acts when being downscaled
 96     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_NEAREST);
 97 
 98     // Mipmap generation - diffent textures for different camera distance
 99     gl.generateMipmap(gl.TEXTURE_2D);
100 
101     // Unbind
102     gl.bindTexture(gl.TEXTURE_2D, null);
103 }

105 /**
106 * Handles JSON file
107 * Creates objects from this file
108 * @param inputJSON JSON file
109 */
110 function handleLoadedJSON(inputJSON) {
111 
112     if(log){
113         console.log("handling");
114     }
115 
116     var i;
117     var j = 0;
118     var textureCounter = 0;
119     var staticObjectsCounter = 0;
120     var materialsCounter;
121 
122     // Loading JSON file
123     for (structure in inputJSON) {
124 
125         // Container
126         if ((inputJSON[structure].type == "geometry_container" || 
127              inputJSON[structure].type == "geometry_container_poly") && 
128              !(parseInt(inputJSON[structure].hierarchy_level) > 0)) {
129 
130             // Poly container
131             if(inputJSON[structure].type == "geometry_container_poly"){
132                 
133                 // Using lightmaps
134                 if(useLightmaps){
135 
136                     try{
137                             var polyID = parseInt(inputJSON[structure].poly_id);
138                             // Load lightmap
139                             lightmaps[polyID] = (new Lightmap(polyID, (lightmapPath + selectedLevel)));
140                     }
141                     catch(error){
142                         ;
143                     }
144                 }
145             }
146             
147             // Dynamic container
148             if (inputJSON[structure]["container_id"] != -1) {
149 
150                 // Create dynamic item object
151                 dynamicObjects[inputJSON[structure]["container_id"]] = new GeometryContainer(inputJSON[structure]);
152             } 
153             // Static container
154             else {
155 
156                 // Create static item object
157                 staticObjects[staticObjectsCounter] = new GeometryContainer(inputJSON[structure]);
158                 staticObjectsCounter++;
159             }
160         } 
161         // Material
162         else if (inputJSON[structure].type == "material") {
163 
164             // Create material object
165             materials["" + inputJSON[structure].name] = new Material(inputJSON[structure]);
166         } 
167         // Level logic
168         else if (inputJSON[structure].type == "level_logic") {
169 
170             // Create logic object
171             logic = new Logic(inputJSON[structure]);
172 
173         }
174         // Light
175         else if(inputJSON[structure].type == "light"){
176 
177             // Create light object
178             lights.push(new Light(inputJSON[structure]));
179         }
180     }
181     
182     var lightning = true;
183     gl.uniform1i(shaderProgram.useLightingUniform, lightning);
184     
185     // Create light
186     var light = {
187         "type" : "light",
188         "name" : "Svetlo 0",
189         "position" : [ logic.centerX, 0, logic.centerZ],
190         "diffuse_color" : [ 0.8, 0.8, 0.8],
191         "coeficients" : [ 0.000000, 0.250000, 0.000000],
192         "max_range" : "5.000000"
193     }
194     lights[0] = new Light( light );
195     
196     // Set lightning
197     if (lightning) {
198         gl.uniform3f(shaderProgram.ambientColorUniform, lights[0].diffuseColor[0], lights[0].diffuseColor[1], lights[0].diffuseColor[2]);
199         gl.uniform3f(shaderProgram.pointLightingLocationUniform, lights[0].position[0], lights[0].position[1], lights[0].position[2]);
200         gl.uniform3f(shaderProgram.pointLightingSpecularColorUniform, lights[0].diffuseColor[0], lights[0].diffuseColor[1], lights[0].diffuseColor[2]);
201         gl.uniform3f(shaderProgram.pointLightingDiffuseColorUniform, lights[0].diffuseColor[0], lights[0].diffuseColor[1], lights[0].diffuseColor[2]);
202     }
203 
204     // Create selected bug texture
205     var selectedBugTexture = {
206 
207       "type" : "material",
208       "name" : "beruska_chosen",
209       "transparent" : "0",
210       "z_buffer_mask" : "1",
211       "z_buffer_test" : "1",
212       "backface_culling" : "0",
213       "diffuse_color" : "1",
214       "specular_color" : "0",
215       "textures" : [ "256_beruska1_chosen.jpg" ]
216     }
217 
218     materials[selectedBugTexture.name] = new Material(selectedBugTexture);
219 
220     // Create floor texture
221     var floorTexture = {
222 
223       "type" : "material",
224       "name" : "floor_texture",
225       "transparent" : "0",
226       "z_buffer_mask" : "1",
227       "z_buffer_test" : "1",
228       "backface_culling" : "0",
229       "diffuse_color" : "1",
230       "specular_color" : "0",
231       "textures" : [ "floor_metal.jpg" ]
232     }
233 
234     materials[floorTexture.name] = new Material(floorTexture);
235     
236     // START DRAWING
237     tick();
238 
239     // Start FPS showing
240     if(showFPSinConsole){
241         if(fpsTimeoutID != 0){
242             clearTimeout(fpsTimeoutID);
243         }
244         fpsTimeoutID = setTimeout(showFPS, 1000);  
245     }
246 
247 }

275 /**
276 * Loads level specified in selectedLevel variable
277 */
278 function loadLevel() {
279     
280     loaded = false;
281 
282     // Loading notification
283     notificationAnimation('loading');
284     $("#progressbar").progressbar({value: 0}).stop().css({'opacity':'0.0'}).animate({'opacity':'1.0'}, 400);
285 
286     // Load level from server
287     var request = new XMLHttpRequest();
288     request.open("GET", levelPath + selectedLevel + levelFileType);
289     // Set handle
290     request.onreadystatechange = function() {
291         
292         if (request.readyState == 4) {
293             
294             // Load JSON file
295             handleLoadedJSON(JSON.parse(request.responseText));
296 
297             // JSON loaded
298             jsonLoaded();
299 
300             // Select Bug
301             logic.selectNextBug();
302 
303             // Do correntions
304             corrections();
305 
306         }
307     }
308 
309     // Make request
310     request.send();
311 }
