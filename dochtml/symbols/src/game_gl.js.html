 43 /**
 44 * Loading of shaders from HTML page
 45 * @param gl WebGL object
 46 * @param id Shader type
 47 * @return Compiled shader
 48 */
 49 function getShader(gl, id) {
 50 
 51     // Shader script loading
 52     var shaderScript = document.getElementById(id);
 53 
 54     if (!shaderScript) {
 55 
 56         return null;
 57     }
 58 
 59     var str = "";
 60     var k = shaderScript.firstChild;
 61     while (k) {
 62 
 63         if (k.nodeType == 3) {
 64 
 65             str += k.textContent;
 66         }
 67         k = k.nextSibling;
 68     }
 69 
 70     // Type of shader
 71     var shader;
 72     if (shaderScript.type == "x-shader/x-fragment") {
 73 
 74         // Creating Fragment shader
 75         shader = gl.createShader(gl.FRAGMENT_SHADER);
 76     } else if (shaderScript.type == "x-shader/x-vertex") {
 77 
 78         // Creating Vertex shader
 79         shader = gl.createShader(gl.VERTEX_SHADER);
 80     } else {
 81         return null;
 82     }
 83 
 84     // Shader Compilation
 85     gl.shaderSource(shader, str);
 86     gl.compileShader(shader);
 87 
 88     if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS)) {
 89 
 90         alert(gl.getShaderInfoLog(shader));
 91         return null;
 92     }
 93 
 94     return shader;
 95 }

 97 /**
 98 * Shader initialization
 99 */
100 function initShaders() {
101 
102     // Loading shaders from HTML
103     var fragmentShader = getShader(gl, "per-fragment-lighting-fs");
104     var vertexShader = getShader(gl, "per-fragment-lighting-vs");
105 
106     // Creating shader program
107     shaderProgram = gl.createProgram();
108     // Attaching Vertex Shader
109     gl.attachShader(shaderProgram, vertexShader);
110     // Attaching fragment shader
111     gl.attachShader(shaderProgram, fragmentShader);
112     gl.linkProgram(shaderProgram);
113 
114     if (!gl.getProgramParameter(shaderProgram, gl.LINK_STATUS)) {
115         alert("Could not initialise shaders");
116     }
117 
118     // Using curretly created shader program
119     gl.useProgram(shaderProgram);
120 
121     // Getting reference to Vertex position attribute
122     shaderProgram.vertexPositionAttribute = gl.getAttribLocation(shaderProgram, "aVertexPosition");
123     gl.enableVertexAttribArray(shaderProgram.vertexPositionAttribute);
124 
125     // Getting reference to Vertex normal attribute
126     shaderProgram.vertexNormalAttribute = gl.getAttribLocation(shaderProgram, "aVertexNormal");
127     gl.enableVertexAttribArray(shaderProgram.vertexNormalAttribute);
128 
129     // Getting reference to Texture coordination attribute
130     shaderProgram.textureCoordAttribute = gl.getAttribLocation(shaderProgram, "aTextureCoord");
131     gl.enableVertexAttribArray(shaderProgram.textureCoordAttribute);
132     
133     // Getting reference to Lightmap coordination attribute
134     shaderProgram.lightmapCoordAttribute = gl.getAttribLocation(shaderProgram, "vLightmapCoord");
135     gl.enableVertexAttribArray(shaderProgram.lightmapCoordAttribute);
136 
137     // Reference to Projection Matrix
138     shaderProgram.pMatrixUniform = gl.getUniformLocation(shaderProgram, "uPMatrix");
139 
140     // Reference to Model-View Matrix
141     shaderProgram.mvMatrixUniform = gl.getUniformLocation(shaderProgram, "uMVMatrix");
142 
143     // Reference to Normal Matrix
144     shaderProgram.nMatrixUniform = gl.getUniformLocation(shaderProgram, "uNMatrix");
145 
146     // Reference to Texture Sampler
147     shaderProgram.samplerUniform = gl.getUniformLocation(shaderProgram, "uSampler");
148 
149     // Reference to Material Specular value
150     shaderProgram.materialShininessUniform = gl.getUniformLocation(shaderProgram, "uMaterialShininess");
151 
152     // Reference to variable used for enabling specular highlights
153     shaderProgram.showSpecularHighlightsUniform = gl.getUniformLocation(shaderProgram, "uShowSpecularHighlights");
154 
155     // Reference to variable used for enabling usage of textures
156     shaderProgram.useTexturesUniform = gl.getUniformLocation(shaderProgram, "uUseTextures");
157 
158     // Reference to variable used for enabling usage of lightning
159     shaderProgram.useLightingUniform = gl.getUniformLocation(shaderProgram, "uUseLighting");
160 
161     // Reference to ambient color value
162     shaderProgram.ambientColorUniform = gl.getUniformLocation(shaderProgram, "uAmbientColor");
163 
164     // Location of Point Light in the scene
165     shaderProgram.pointLightingLocationUniform = gl.getUniformLocation(shaderProgram, "uPointLightingLocation");
166 
167     // Point light specular color
168     shaderProgram.pointLightingSpecularColorUniform = gl.getUniformLocation(shaderProgram, "uPointLightingSpecularColor");
169 
170     // Point light diffuse color
171     shaderProgram.pointLightingDiffuseColorUniform = gl.getUniformLocation(shaderProgram, "uPointLightingDiffuseColor");
172     shaderProgram.alphaUniform = gl.getUniformLocation(shaderProgram, "uAlpha");
173     
174     // These variables are not used - used for more lights in the scene
175     //shaderProgram.pointLightningLocationUniform1 = gl.getUniformLocation(shaderProgram, "uPointLightingLocationArray");
176     //shaderProgram.lightCount = gl.getUniformLocation(shaderProgram, "lightCount");
177     
178     // Lightmap sampler
179     shaderProgram.lightmapSamplerUniform = gl.getUniformLocation(shaderProgram, "uLightmapSampler");
180 
181     // Enabling lightmaps
182     shaderProgram.useLightmapUniform = gl.getUniformLocation(shaderProgram, "uUseLightmap");
183 }

185 /**
186 * Initialization of main WebGL context
187 * @param Canvas Reference to canvas DOM object
188 */
189 function initGL(canvas) {
190     
191     // Not all of browser are supporting WebGL
192     try {
193 
194         // Gettings context
195         gl = canvas.getContext("experimental-webgl");
196 
197         // Settings draw buffer width and height
198         gl.viewportWidth = canvas.width;
199         gl.viewportHeight = canvas.height;
200 
201         glInitialized = true;
202     } 
203     catch (e) {}
204 
205     // WebGL cannot be initialised
206     if (!gl) {
207 
208         alert("Could not initialise WebGL. Use diffent browser or upgrade your graphic hardware.");
209     }
210 }

212 /**
213 * Pushes mvMatrix into stack
214 */
215 function mvPushMatrix() {
216 
217     var copy = mat4.create();
218     mat4.set(mvMatrix, copy);
219     mvMatrixStack.push(copy);
220 }
221 
222 /**
223 * Pops mvMatrix from stack
224 */
225 function mvPopMatrix() {
226 
227     if (mvMatrixStack.length == 0) {
228 
229         if(log){
230             console.log("Invalid popMatrix");
231         }
232         //throw "Invalid popMatrix!";
233     }
234     mvMatrix = mvMatrixStack.pop();
235 }

237 /**
238 * Loads mvMatrix, pMatrix and normalMatrix into shaders
239 */
240 function setMatrixUniforms() {
241 
242     gl.uniformMatrix4fv(shaderProgram.pMatrixUniform, false, pMatrix);
243     gl.uniformMatrix4fv(shaderProgram.mvMatrixUniform, false, mvMatrix);
244 
245     var normalMatrix = mat3.create();
246     mat4.toInverseMat3(mvMatrix, normalMatrix);
247     mat3.transpose(normalMatrix);
248     gl.uniformMatrix3fv(shaderProgram.nMatrixUniform, false, normalMatrix);
249 }
